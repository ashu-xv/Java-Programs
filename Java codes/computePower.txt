public class RecursivePower {

    public static double computePower(double base, int exponent) {
        if (exponent < 0) {
            throw new IllegalArgumentException("Exponent should be non-negative.");
        }

        // Base case: any number raised to the power of 0 is 1
        if (exponent == 0) {
            return 1;
        }

        // Recursive case: multiply the base by itself (exponent - 1) times
        return base * computePower(base, exponent - 1);
    }

    public static void main(String[] args) {
        double base = 2.5; // Replace with the desired base
        int exponent = 3; // Replace with the desired exponent

        double result = computePower(base, exponent);

        System.out.println(base + " raised to the power of " + exponent + " is: " + result);
    }
}
