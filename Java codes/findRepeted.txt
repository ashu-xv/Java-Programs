public class RecursiveSearch {

     public static int findFirstRepeatedElement(int[] array) {
        Arrays.sort(array);

        for (int i = 0; i < array.length - 1; i++) {
            if (array[i] == array[i + 1]) {
                return array[i]; // Found the first repeated element
            }
        }

        return -1; // No repeated element found
    }

    public static void main(String[] args) {
        int[] array = {3, 1, 4, 2, 3, 5, 1};

        int linearResult = findFirstRepeatedElement(array);
        if (linearResult != -1) {
            System.out.println("First Repeated Element (Linear Search): " + linearResult);
        } else {
            System.out.println("No repeated elements found in the array (Linear Search).");
        }

      
    }
}
