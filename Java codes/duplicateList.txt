import java.util.ArrayList;
import java.util.List;

public class PrintDuplicatesInList {

    public static void printDuplicates(List<Integer> list) {
        System.out.println("Duplicate Elements in the List:");

        for (int i = 0; i < list.size() - 1; i++) {
            for (int j = i + 1; j < list.size(); j++) {
                if (list.get(i).equals(list.get(j))) {
                    System.out.println(list.get(i));
                    break; // Print the duplicate once and move to the next element
                }
            }
        }
    }

    public static void main(String[] args) {
        List<Integer> myList = new ArrayList<>(List.of(3, 1, 4, 2, 3, 5, 1, 4, 7, 8, 5));

        printDuplicates(myList);
    }
}
