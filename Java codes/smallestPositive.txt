public class SmallestPositiveMissing {

    public static int findSmallestPositiveMissing(int[] nums) {
        int n = nums.length;

        // Rearrange the array
        for (int i = 0; i < n; i++) {
            while (nums[i] > 0 && nums[i] <= n && nums[nums[i] - 1] != nums[i]) {
                // Swap nums[i] with nums[nums[i] - 1]
                int temp = nums[i];
                nums[i] = nums[temp - 1];
                nums[temp - 1] = temp;
            }
        }

        // Find the first position where the element is not at its correct position
        for (int i = 0; i < n; i++) {
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }

        // If all elements are at their correct positions, the missing number is n + 1
        return n + 1;
    }

    public static void main(String[] args) {
        int[] nums = {3, 4, -1, 1};
        int result = findSmallestPositiveMissing(nums);

        System.out.println("Smallest positive missing number: " + result);
    }
}
